```js
class QueueByStack {
  constructor() {
    this.in = [];
    this.out = [];
  }
  push(x) {
    this.in.push(x);
  }

  //返回队头元素
  peek() {
    if (this.out.length == 0) {
      while (this.in.length > 0) {
        this.out.push(this.in.pop());
      }
    }
    return this.out[this.out.length - 1];
  }

  pop() {
    this.peek(); //保证out 不为空
    return this.out.pop();
  }

  empty() {
    return this.in.length == 0 && this.out.length == 0;
  }
}
```
